# DigiLocker API Test Raw XML Bodies (for manual copy-paste into Postman)

# Example 1 (Imliakum GPF)
<?xml version="1.0" encoding="UTF-8"?>
<PullURIRequest xmlns="http://tempuri.org/" ver="3.0" ts="{{timestamp}}" txn="test-txn" orgId="{{issuer_id}}" keyhash="{{keyhash}}" format="both">
  <DocDetails>
    <DocType>GPF</DocType>
    <DigiLockerId>dl-test</DigiLockerId>
    <FullName>Imliakum</FullName>
    <DOB>01-01-1980</DOB>
    <UDF1>AUTH1001</UDF1>
  </DocDetails>
</PullURIRequest>

# 1. Copy the desired XML block.
# 2. In Postman, set Body type to 'raw' and 'XML'.
# 3. Paste the XML and replace {{timestamp}}, {{issuer_id}}, and {{keyhash}} with actual values or Postman variables.
